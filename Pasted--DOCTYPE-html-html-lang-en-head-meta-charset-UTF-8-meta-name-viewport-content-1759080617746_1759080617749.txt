                li.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" class="text-green-500 mr-2 flex-shrink-0"><polyline points="20 6 9 17 4 12"></polyline></svg><span>${skill}</span>`;
                skillsList.appendChild(li);
            });

            // Populate syllabus
            syllabusList.innerHTML = ''; // Clear previous syllabus
            data.syllabus.forEach(item => {
                const div = document.createElement('div');
                div.className = 'p-4 border rounded-lg bg-gray-50';
                div.innerHTML = `
                    <p class="font-semibold text-brand-primary">Weeks ${item.week}: ${item.focus}</p>
                    <p class="text-gray-700 mt-1"><span class="font-semibold">Topics:</span> ${item.topics}</p>
                    <p class="text-sm text-gray-500 mt-2"><span class="font-semibold">Resources:</span> ${item.resources}</p>
                `;
                syllabusList.appendChild(div);
            });
            
             // Populate projects
            projectsList.innerHTML = ''; // Clear previous projects
            const projectDiv = document.createElement('div');
            projectDiv.className = 'p-4 bg-blue-50 rounded-lg border border-blue-200';
            projectDiv.innerHTML = `<p class="text-gray-700 font-medium">${data.project}</p>`;
            projectsList.appendChild(projectDiv);


            // Switch views
            mainContent.classList.add('hidden');
            courseDetailPage.classList.remove('hidden');
            window.scrollTo(0, 0); // Scroll to top
        }

        function hideCourseDetails() {
            mainContent.classList.remove('hidden');
            courseDetailPage.classList.add('hidden');
            const coursesSection = document.getElementById('courses');
            if (coursesSection) {
                coursesSection.scrollIntoView({ behavior: 'smooth' });
            }
        }

        function goToHome() {
            mainContent.classList.remove('hidden');
            courseDetailPage.classList.add('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
